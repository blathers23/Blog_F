        cerr << "getElement(): Indices are out of range" << endl;
        return 0;

    size_t byte_id = index / 8;
    size_t bit_id = index % 8;
    unsigned char mask = (1 << bit_id);

        cerr << "setElement(): Indices are out of range" << endl;
        return false;

    size_t byte_id = index / 8;
    size_t bit_id = index % 8;
    unsigned char mask = (1 << bit_id);

    if (value) {
        data[byte_id] |= mask;
    } else {
        data[byte_id] &= ~mask;
    }

    return true;
