double myPow(double x, int n) {
    double temp;

    double Pow(double x, int n) {
        if (n != 1) {
            if (n % 2 != 0) {
                temp *= x;
            }
            return Pow(x * x, n / 2);
        } else {
            return x * temp;
        }
    }

    if (n < -1) {
        x = 1 / x;
        if (n % 2 != 0) {
            temp = x;
        }
        return Pow(x * x, -(n / 2));	// è§„é¿intå–è´Ÿè¶…å‡ºè½¬èŒƒå›´
    } else if (n == 0) {
        return 1;
    } else if (n == -1) {

    } else {
        temp = 1;
        return Pow(x, n);
    }
}
