double myPow(double x, int n) {
    double temp;

    double Pow(double x, int n) {
        if (n != 1) {
            if (n % 2 != 0) {
                temp *= x;
            }
            return Pow(x * x, n / 2);
        } else {
            return x * temp;
        }
    }

    if (n < -1) {
        x = 1 / x;
        if (n % 2 != 0) {
            temp = x;
        }
        return Pow(x * x, -(n / 2));	// è§„é¿intå–è´Ÿè¶…å‡ºè½¬èŒƒå›´
    } else if (n == 0) {
        return 1;
    } else if (n == -1) {
        return 1 / x;
    } else {
        temp = 1;
        return Pow(x, n);
    }
}

double myPow(double x, int m) {
    long long n = m;

    double Pow(double x, long long n) {
        if (n != 0) {
            return  n % 2 == 0 ? Pow(x * x, n / 2) : Pow(x * x, n / 2) * x;
        } else {
            return 1.0;
        }
    }

    return n >= 0 ? Pow(x, n) : Pow(1 / x, -n);
}
