class Solution:
	def maxArea(self, height: List[int]) -> int:
	maxS = 0
	for r in range(1, len(height)):
		for l in range(0, len(height)):
			maxS = max(maxS, (r - l) * min(height[r], height[l]))
        
	return maxS

int maxArea(int* height, int heightSize){
    int maxS = 0;
    for (int r = 1; r < heightSize; r++) {
        bool Higher = false;
        int Highest = height[0];
        for (int l = 0; l < heightSize; l++) {
            if (height[l] <= height[r]) {
                if (height[l] >= Highest) {
                    maxS = fmax(maxS, (r - l) * fmin(height[l], height[r]));
                    Highest = height[l];
                }
            } else {
                if (!Higher) {
                    maxS = fmax(maxS, (r - l) * fmin(height[l], height[r]));
                    Higher = true;
                }
            }
        }
    }   
    return maxS;
}

int maxArea(int* height, int heightSize){
    int minH = 0;
    int maxS = 0;
    int maxS_ = 0;
    int l = 0;
    int r = heightSize - 1;
    int maxL = height[l];
    int maxR = height[r];

    while (r > l) {
        if (maxL > maxR) {		// ä¸¤æ•°ä¹‹ä¸­å–å…¶å°
            minH = maxR;
        } else {
            minH = maxL;
        }
        maxS_ = (r - l) * minH;	// ä¸´æ—¶ç››æ°´é‡
        if (maxS_ > maxS) {		// ä¸¤æ•°ä¹‹ä¸­å–å…¶å¤§
            maxS = maxS_;
        }
        while (r > l){
            if (height[r] >= height[l]) {
                l++;
                if (height[l] > maxL) {
                    maxL = height[l];
                    break;
                } 
            } else {
                r--;
                if (height[r] > maxR) {
                    maxR = height[r];
                    break;
                }
            }
        }
    }
    return maxS;
}
